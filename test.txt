<script>

function showPanel(panelId){
    const panels = document.querySelectorAll('.panel');
    panels.forEach(p => p.classList.remove('active'));

    setTimeout(()=>{
        document.getElementById(panelId).classList.add('active');
    },200);
}

document.getElementById("complaintForm").addEventListener("submit", function(e){
    e.preventDefault();

    if(!otpVerified){
        alert("Please verify OTP before submitting.");
        return;
    }


    let complaints = JSON.parse(localStorage.getItem("complaints")) || [];

    let complaint = {
        id: Date.now(),
        category: category.value,
        description: description.value,
        contact: phone.value,
        status: "Pending"
    };

    complaints.push(complaint);
    localStorage.setItem("complaints", JSON.stringify(complaints));

    document.getElementById("successMsg").style.display="block";
    this.reset();
});

function trackComplaint(){
    let phone = document.getElementById("trackPhone").value;
    let complaints = JSON.parse(localStorage.getItem("complaints")) || [];
    let results = document.getElementById("trackResults");
    results.innerHTML="";

    let filtered = complaints.filter(c => c.contact === phone);

    if(filtered.length === 0){
        results.innerHTML="No complaints found.";
        return;
    }

    filtered.forEach(c=>{
        results.innerHTML += `
        <div style="background:#111;padding:15px;border-radius:10px;margin-top:10px;">
            <strong>${c.category}</strong><br>
            ${c.description}<br>
            Status: ${c.status}
        </div>
        `;
    });
}

let generatedOTP = null;
let otpVerified = false;

function sendOTP(){
    const phone = document.getElementById("phone").value;

    if(!phone || phone.length !== 10){
        alert("Enter valid 10-digit phone number.");
        return;
    }

    generatedOTP = Math.floor(1000 + Math.random()*9000).toString();
    otpVerified = false;

    alert("Demo OTP: " + generatedOTP);

    document.getElementById("otpSection").style.display = "block";
    document.getElementById("otpStatus").innerText = "";
}

function verifyOTP(){
    const enteredOTP = document.getElementById("otpInput").value;

    if(enteredOTP === generatedOTP){
        otpVerified = true;
        document.getElementById("otpStatus").style.color = "lightgreen";
        document.getElementById("otpStatus").innerText = "OTP Verified ✓";
    } else {
        otpVerified = false;
        document.getElementById("otpStatus").style.color = "red";
        document.getElementById("otpStatus").innerText = "Invalid OTP";
    }
}
